**Q16:** Write a function that converts a string to snake_case (lowercase with underscores). Input: `"FirstName"` → `"first_name"`.

<details>
<summary>Solution</summary>

```python
def to_snake_case(name):
    result = []
    for i, ch in enumerate(name):
        if ch.isupper() and i > 0:
            result.append('_')
        result.append(ch.lower())
    return ''.join(result)

print(to_snake_case("FirstName"))     # 'first_name'
print(to_snake_case("HTTPServer"))    # 'http_server'
```
</details>


**Q23:** Find the second largest element in a list without sorting.

```python
def second_largest(lst):
    if len(lst) < 2:
        return None
    first = second = float('-inf')
    for n in lst:
        if n > first:
            second = first
            first = n
        elif n > second and n != first:
            second = n
    return second if second != float('-inf') else None

print(second_largest([3,1,4,1,5,9,2]))  # 5
```
</details>

```python
def product_except_self(nums):
    n = len(nums)
    result = [1] * n
    left = 1
    for i in range(n):
        result[i] = left
        left *= nums[i]
    right = 1
    for i in range(n-1, -1, -1):
        result[i] *= right
        right *= nums[i]
    return result

print(product_except_self([1,2,3,4]))  # [24,12,8,6]
```

**Q25:** Given two sorted lists, merge them into one sorted list without using `sort()`.

<details>
<summary>Solution</summary>

```python
def merge_sorted(a, b):
    i, j = 0, 0
    result = []
    while i < len(a) and j < len(b):
        if a[i] < b[j]:
            result.append(a[i])
            i += 1
        else:
            result.append(b[j])
            j += 1
    result.extend(a[i:])
    result.extend(b[j:])
    return result

print(merge_sorted([1,3,5], [2,4,6]))  # [1,2,3,4,5,6]
```

**Q40:** Find the longest consecutive sequence in an unsorted list using a set (O(n)). Example `[100,4,200,1,3,2]` → `4` (sequence 1,2,3,4).

<details>
<summary>Solution</summary>

```python
def longest_consecutive(nums):
    num_set = set(nums)
    longest = 0
    for n in num_set:
        if n - 1 not in num_set:  # start of a sequence
            length = 1
            while n + length in num_set:
                length += 1
            longest = max(longest, length)
    return longest

print(longest_consecutive([100,4,200,1,3,2]))  # 4
```
</details>

**Q65:** Flatten a nested dictionary into dot‑separated keys. Example: `{"a":{"b":1}}` → `{"a.b":1}`.

<details>
<summary>Solution</summary>

```python
def flatten_dict(d, parent_key='', sep='.'):
    items = []
    for k, v in d.items():
        new_key = f"{parent_key}{sep}{k}" if parent_key else k
        if isinstance(v, dict):
            items.extend(flatten_dict(v, new_key, sep=sep).items())
        else:
            items.append((new_key, v))
    return dict(items)

print(flatten_dict({"a":{"b":{"c":1}}}))  # {'a.b.c':1}
```
</details>

**Q66:** Deep merge two nested dictionaries (recursively combine, not overwrite).

<details>
<summary>Solution</summary>

```python
def deep_merge(a, b):
    for k in b:
        if k in a and isinstance(a[k], dict) and isinstance(b[k], dict):
            deep_merge(a[k], b[k])
        else:
            a[k] = b[k]
    return a
```

**Q68:** Build a nested dictionary from a list of paths (`["home/user/docs", "home/user/pics"]`) where leaf values are `None`.

```python
def build_nested(paths):
    root = {}
    for path in paths:
        parts = path.split('/')
        current = root
        for part in parts[:-1]:
            current = current.setdefault(part, {})
        current[parts[-1]] = None
    return root

paths = ["home/user/docs", "home/user/pics", "var/log"]
```

**Q77:** Write a function that removes duplicates from a list of unhashable items (e.g., dictionaries) by converting to a string representation.

```python
def remove_duplicate_dicts(lst):
    seen = set()
    result = []
    for d in lst:
        # Use tuple of sorted items for hashable representation
        repr_d = tuple(sorted(d.items()))
        if repr_d not in seen:
            seen.add(repr_d)
            result.append(d)
    return result

lst = [{"a":1,"b":2}, {"b":2,"a":1}, {"c":3}]
```

## 12. Mini Project: Contact Manager (Practice All Topics)

**Q79:** Build a contact manager using a nested dictionary where each contact has name, phone, email, and tags (set). Implement:
- Add contact
- Search by name
- Search by tag
- Delete contact
- List all contacts sorted by name

<details>
<summary>Solution skeleton</summary>

```python
contacts = {}  # name -> {"phone":..., "email":..., "tags": set()}

def add_contact(name, phone, email, tags=None):
    contacts[name] = {
        "phone": phone,
        "email": email,
        "tags": set(tags) if tags else set()
    }

def search_by_name(query):
    return {name: info for name, info in contacts.items() if query.lower() in name.lower()}

def search_by_tag(tag):
    return {name: info for name, info in contacts.items() if tag in info["tags"]}

def delete_contact(name):
    return contacts.pop(name, None) is not None

def list_all():
    for name in sorted(contacts.keys()):
        info = contacts[name]
        print(f"{name}: {info['phone']}, {info['email']}, tags: {info['tags']}")

# Example usage
add_contact("Alice", "123-456", "alice@example.com", {"friend", "work"})
add_contact("Bob", "789-012", "bob@example.com", {"friend"})
print(search_by_tag("friend"))
delete_contact("Bob")
list_all()
```
